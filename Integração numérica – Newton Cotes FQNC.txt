def simpson_38(t, v, h):
    """
    Calcula a integral numérica utilizando a Regra 3/8 de Simpson.
    Focada em estruturas nativas para atender aos requisitos do projeto.
    """
    # Validação do número de pontos necessários para a regra 3/8 simples (4 pontos)
    if len(t) != 4 or len(v) != 4:
        raise ValueError("A Regra 3/8 de Simpson simples requer exatamente 4 pontos.")
        
    # Aplicação direta da fórmula matemática de Newton-Cotes
    # Peso 1 para as extremidades e peso 3 para os pontos internos
    soma = v[0] + 3 * v[1] + 3 * v[2] + v[3]
    
    # Cálculo do valor final da integral
    integral = (3 * h / 8) * soma
    return integral

# --- DADOS DO PROBLEMA (PÁGINA 10) ---
# t (em segundos): [0, 2, 4, 6] com h = 2
# v (banda em MB/s): [10, 15, 12, 8]
tempo = [0, 2, 4, 6]
velocidade = [10, 15, 12, 8]
h_constante = 2

# Execução do algoritmo nativo
total_mb = simpson_38(tempo, velocidade, h_constante)

# --- RETORNO DA RESPOSTA DO ALGORITMO ---
print("=" * 50)
print("   RESULTADO DA INTEGRAÇÃO NUMÉRICA (NEWTON-COTES)   ")
print("=" * 50)
print(f"Dados de Tempo (t):       {tempo} s")
print(f"Dados de Velocidade (v):  {velocidade} MB/s")
print(f"Espaçamento constante (h): {h_constante}")
print("-" * 50)
print(f"Total de dados transferidos: {total_mb:.2f} MB")
print("=" * 50)
